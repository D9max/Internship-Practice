 //print 1-10 numbers using for loop

Begin
   number i=1;
  for(i=1;i<=10;i++)               
  {
        print(i)
  }  
 
end

Begin
   number i=1;
  while i<=10              
  {
        print(i)
        i=i+1
  }  
 
end
